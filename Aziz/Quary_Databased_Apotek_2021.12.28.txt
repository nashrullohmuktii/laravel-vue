SELECT nama, tb_resep.id_obat, nama_obat, jenis_obat, COUNT(*)
FROM `tb_resep`
JOIN tb_obat ON tb_resep.id_obat=tb_obat.id_obat
GROUP BY nama_obat
ORDER BY nama_obat ASC

SELECT nama, tb_resep.id_obat, nama_obat, jenis_obat, COUNT(*)
FROM `tb_resep`
RIGHT JOIN tb_obat ON tb_resep.id_obat=tb_obat.id_obat
WHERE NOT nama LIKE '%NULL%'
GROUP BY nama_obat
ORDER BY nama_obat ASC

SELECT nama, tb_resep.id_obat, nama_obat, jenis_obat, COUNT(*)
FROM `tb_resep`
LEFT JOIN tb_obat ON tb_resep.id_obat=tb_obat.id_obat
GROUP BY nama_obat
ORDER BY nama_obat ASC
